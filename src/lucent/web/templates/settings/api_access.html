Example: REST

# Search memories
curl "{{ base_url }}/api/search?query=python" \
  -H "Authorization: Bearer hs_your_api_key_here"

# Create a memory
curl -X POST "{{ base_url }}/api/memories" \
  -H "Authorization: Bearer hs_your_api_key_here" \
  -H "Content-Type: application/json" \
  -d '{"type": "technical", "content": "How to fix X..."}'
